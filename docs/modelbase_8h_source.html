327template <StateType num_states>
328void cmaple::ModelBase::updateMutationMat() {
329 // update Mutation matrix regarding the pseudo muation count
330 updateMutMatbyMutCount();
331
332 // compute the total rate regarding the root freqs
333 RealNumType total_rate = 0;
334 total_rate -= dotProduct<num_states>(root_freqs, diagonal_mut_mat);
335
336 // inverse total_rate
337 total_rate = 1.0 / total_rate;
338
339 // normalize the mutation_mat
340 RealNumType* mutation_mat_row = mutation_mat;
341 RealNumType* freqi_freqj_qij_row = freqi_freqj_qij;
342 for (StateType i = 0; i < num_states_; ++i, mutation_mat_row += num_states_,
343 freqi_freqj_qij_row += num_states_) {
344 for (StateType j = 0; j < num_states_; ++j) {
345 mutation_mat_row[j] *= total_rate;
346 // mutation_mat_row[j] /= total_rate;
347
348 // update freqi_freqj_qij
349 if (i != j) {
350 freqi_freqj_qij_row[j] =
351 root_freqs[i] * inverse_root_freqs[j] * mutation_mat_row[j];
352 // root_freqs[i] / root_freqs[j] * mutation_mat_row[j];
353 } else {
354 freqi_freqj_qij_row[j] = mutation_mat_row[j];
355 }
356
357 // update the transposed mutation matrix
358 transposed_mut_mat[row_index[j] + i] = mutation_mat_row[j];
359 }
360
361 // update diagonal
362 diagonal_mut_mat[i] = mutation_mat_row[i];
363 }
364
365 // pre-compute matrix to speedup
366 RealNumType* transposed_mut_mat_row = transposed_mut_mat;
367 RealNumType* freq_j_transposed_ij_row = freq_j_transposed_ij;
368
369 for (StateType i = 0; i < num_states_; ++i,
370 transposed_mut_mat_row += num_states_,
371 freq_j_transposed_ij_row += num_states_) {
372 setVecByProduct<num_states>(freq_j_transposed_ij_row, root_freqs,
373 transposed_mut_mat_row);
374 }
375}

377template <StateType num_states>
378auto cmaple::ModelBase::updateMutationMatEmpiricalTemplate()
379 -> bool {
380 bool update = false;
381
382 if (!fixed_params) {
383 // clone the current mutation matrix
384 RealNumType* tmp_mut_mat = new RealNumType[row_index[num_states_]];
385 memcpy(tmp_mut_mat, mutation_mat,
386 row_index[num_states_] * sizeof(RealNumType));
387
388 // update the mutation matrix regarding the pseu_mutation_count
389 updateMutationMat<num_states>();
390
391 // set update = true if the mutation matrix changes more than a threshold
392 const RealNumType change_thresh = 1e-3;
393 RealNumType sum_change = 0;
394 RealNumType* tmp_mut_mat_ptr = tmp_mut_mat;
395 RealNumType* mutation_mat_ptr = mutation_mat;
396 for (StateType i = 0; i < num_states_;
397 ++i, tmp_mut_mat_ptr += num_states_, mutation_mat_ptr += num_states_) {
398 for (StateType j = 0; j < num_states_; ++j) {
399 sum_change += fabs(tmp_mut_mat_ptr[j] - mutation_mat_ptr[j]);
400 }
401
402 sum_change -= fabs(tmp_mut_mat_ptr[i] - mutation_mat_ptr[i]);
403 }
404
405 update = sum_change > change_thresh;
406
407 // delete tmp_diagonal_mutation_mat
408 delete[] tmp_mut_mat;
409 }
410
411 // return update
412 return update;
413}
