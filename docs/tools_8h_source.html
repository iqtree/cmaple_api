1111 template<class T1, class T2>
1112 void quicksort(T1* arr, int left, int right, T2* arr2 = NULL) {
1113 if (left > right) return;
1114 ASSERT(left <= right);
1115 int i = left, j = right;
1116 T1 pivot = arr[(left + right) / 2];
1117
1118 /* partition */
1119 while (i <= j) {
1120 while (arr[i] < pivot)
1121 ++i;
1122 while (arr[j] > pivot)
1123 --j;
1124 if (i <= j) {
1125 T1 tmp = arr[i];
1126 arr[i] = arr[j];
1127 arr[j] = tmp;
1128 if (arr2) {
1129 T2 tmp2 = arr2[i];
1130 arr2[i] = arr2[j];
1131 arr2[j] = tmp2;
1132 }
1133 ++i;
1134 --j;
1135 }
1136 };
1137
1138 /* recursion */
1139 if (left < j)
1140 quicksort(arr, left, j, arr2);
1141 if (i < right)
1142 quicksort(arr, i, right, arr2);
1143 }
