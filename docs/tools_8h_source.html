1023 template<class T1, class T2>
1024 void quicksort(T1* arr, int left, int right, T2* arr2 = NULL) {
1025 if (left > right) return;
1026 ASSERT(left <= right);
1027 int i = left, j = right;
1028 T1 pivot = arr[(left + right) / 2];
1029
1030 /* partition */
1031 while (i <= j) {
1032 while (arr[i] < pivot)
1033 ++i;
1034 while (arr[j] > pivot)
1035 --j;
1036 if (i <= j) {
1037 T1 tmp = arr[i];
1038 arr[i] = arr[j];
1039 arr[j] = tmp;
1040 if (arr2) {
1041 T2 tmp2 = arr2[i];
1042 arr2[i] = arr2[j];
1043 arr2[j] = tmp2;
1044 }
1045 ++i;
1046 --j;
1047 }
1048 };
1049
1050 /* recursion */
1051 if (left < j)
1052 quicksort(arr, left, j, arr2);
1053 if (i < right)
1054 quicksort(arr, i, right, arr2);
1055 }
