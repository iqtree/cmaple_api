1094 template<class T1, class T2>
1095 void quicksort(T1* arr, int left, int right, T2* arr2 = NULL) {
1096 if (left > right) return;
1097 ASSERT(left <= right);
1098 int i = left, j = right;
1099 T1 pivot = arr[(left + right) / 2];
1100
1101 /* partition */
1102 while (i <= j) {
1103 while (arr[i] < pivot)
1104 ++i;
1105 while (arr[j] > pivot)
1106 --j;
1107 if (i <= j) {
1108 T1 tmp = arr[i];
1109 arr[i] = arr[j];
1110 arr[j] = tmp;
1111 if (arr2) {
1112 T2 tmp2 = arr2[i];
1113 arr2[i] = arr2[j];
1114 arr2[j] = tmp2;
1115 }
1116 ++i;
1117 --j;
1118 }
1119 };
1120
1121 /* recursion */
1122 if (left < j)
1123 quicksort(arr, left, j, arr2);
1124 if (i < right)
1125 quicksort(arr, i, right, arr2);
1126 }
